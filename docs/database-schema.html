USERS

Campo Tipo Constraints Descricao
id PK INTEGER AUTO_INCREMENT, NOT NULL Identificador unico do usuario
name TEXT NOT NULL Nome completo do usuario
email UNIQUE TEXT UNIQUE, NOT NULL Email unico para login
password_hash TEXT NOT NULL Senha criptografada com bcrypt
created_at DATETIME NOT NULL, DEFAULT CURRENT_TIMESTAMP Data de criacao do registro
updated_at DATETIME NOT NULL, DEFAULT CURRENT_TIMESTAMP Data da ultima atualizacao

TASKS

Campo Tipo Constraints Descricao
id PK INTEGER AUTO_INCREMENT, NOT NULL Identificador unico da tarefa
title TEXT NOT NULL Titulo da tarefa
description TEXT NOT NULL Descricao detalhada da tarefa
status TEXT NOT NULL, DEFAULT 'PENDING' Status da tarefa (PENDING, COMPLETED)
user_id FK INTEGER NOT NULL, REFERENCES users(id) ID do usuario proprietario
created_at DATETIME NOT NULL, DEFAULT CURRENT_TIMESTAMP Data de criacao do registro
updated_at DATETIME NOT NULL, DEFAULT CURRENT_TIMESTAMP Data da ultima atualizacao
